import { LayerType, Layer } from './layer';
import { SpatialReference } from '../spatial-reference';
/**
 * 高程模式
 *
 * @date 10/12/2021
 * @export
 * @enum {number}
 */
export var ElevationMode;
(function (ElevationMode) {
    /**
     * 贴地形表面
     */
    ElevationMode[ElevationMode["OnTheGround"] = 0] = "OnTheGround";
    /**
     * 相对于地形表面
     */
    ElevationMode[ElevationMode["RelativeToGround"] = 1] = "RelativeToGround";
    /**
     * 绝对高度
     */
    ElevationMode[ElevationMode["AbsoluteHeight"] = 2] = "AbsoluteHeight";
    /**
     * 相对于场景
     */
    ElevationMode[ElevationMode["RelativeToScene"] = 3] = "RelativeToScene";
})(ElevationMode || (ElevationMode = {}));
/**
 * 要素表达式信息
 *
 * @date 10/12/2021
 * @export
 * @class FeatureExpressionInfo
 */
export class FeatureExpressionInfo {
    /**
     * 用于确定要素高度的表达式。设置了该表达式后，几何自身的z坐标将会忽略。当高程模式为OnTheGround时，
     * 该属于无效。
     *
     * @date 10/12/2021
     * @memberof FeatureExpressionInfo
     */
    expression = '';
}
/**
 * 距离单位
 *
 * @date 10/12/2021
 * @export
 * @enum {number}
 */
export var LengthUnit;
(function (LengthUnit) {
    /**
     * 米
     */
    LengthUnit[LengthUnit["Meters"] = 0] = "Meters";
    /**
     * 千米
     */
    LengthUnit[LengthUnit["Kilometers"] = 1] = "Kilometers";
})(LengthUnit || (LengthUnit = {}));
/**
 * 高程信息类
 * 描述要素在Z方向上如何放置。
 * @date 10/12/2021
 * @export
 * @class ElevationInfo
 */
export class ElevationInfo {
    /**
     * 高程模式
     *
     * @date 10/12/2021
     * @memberof ElevationInfo
     */
    mode = ElevationMode.OnTheGround;
    /**
     * 高程偏移。默认为零，默认单位为米。
     *
     * @date 10/12/2021
     * @memberof ElevationInfo
     */
    offset = 0;
    /**
     * 要素表达式信息
     *
     * @date 10/12/2021
     * @memberof ElevationInfo
     */
    featureExpressionInfo;
    /**
     * offset和featureExpression的单位
     *
     * @date 10/12/2021
     * @memberof ElevationInfo
     */
    unit = LengthUnit.Meters;
}
/**
 * 要素图层
 *
 * @author Yuanye Ma
 * @date 09/12/2021
 * @export
 * @abstract
 * @class FeatureLayer
 * @extends {Layer}
 */
export class FeatureLayer extends Layer {
    /**
     * Creates an instance of FeatureLayer.
     * @date 22/03/2021
     * @param {Record<string, any>} [properties]
     * @memberof FeatureLayer
     */
    constructor(properties) {
        super(properties);
        this.type = LayerType.Feature;
        if (!properties)
            return;
        if (properties.maxScale)
            this.maxScale = properties.maxScale;
        if (properties.minScale)
            this.minScale = properties.minScale;
        if (properties.tokenKey)
            this.tokenKey = properties.tokenKey;
        if (properties.tokenValue)
            this.tokenValue = properties.tokenValue;
    }
    /**
     * 图层显示的最大比例尺:在地图视图中超出该比例尺,图层将不再显示。
     * 默认值：0. 0表示该图层的显示不受最大比例尺的限制，可以无限放大。
     * maxScale < minScale
     * sample
     * The layer will not be visible when the view is zoomed in beyond a scale of 1:2,000
     * layer.maxScale = 2000;
     *
     * @date 22/03/2021
     * @memberof FeatureLayer
     */
    maxScale = 0;
    /**
     * 图层显示的最小比例尺:在地图视图中超出该比例尺,图层将不再显示。
     * 默认值：0. 0表示该图层的显示不受最小比例尺的限制，可以无限缩小。
     *
     * @date 22/03/2021
     * @memberof FeatureLayer
     */
    minScale = 0;
    /**
     * token的参数名
     *
     * @date 22/03/2021
     * @memberof FeatureLayer
     */
    tokenKey = 'token';
    /**
     * token的参数值
     *
     * @date 22/03/2021
     * @memberof FeatureLayer
     */
    tokenValue = '';
    /**
     * 空间参数信息
     *
     * @date 28/04/2021
     * @memberof FeatureLayer
     */
    spatialReference = new SpatialReference();
    /**
     * SQL查询条件，用于过滤要素
     *
     * @date 09/12/2021
     * @memberof FeatureLayer
     */
    definitionExpression = '';
    /**
     * 高程信息
     *
     * @date 10/12/2021
     * @type {ElevationInfo}
     * @memberof FeatureLayer
     */
    elevationInfo = new ElevationInfo();
    /**
     * 几何类型
     *
     * @value Lin:线、Pnt:点、Net:网络类
     * @date 10/12/2021
     * @memberof FeatureLayer
     */
    geomType = '';
    /**
     * 服务器端返回的要素的字段列表。
     * 如果没有设置，默认只返回FID字段。
     * 返回所有字段，用["*"]。
     * 指定段:["name","GDP_2016","GDP_2017"]
     *
     * @date 10/12/2021
     * @type {string[]}
     * @memberof FeatureLayer
     */
    outFields;
    /**
     * 是否支持拾取
     *
     * @date 10/12/2021
     * @memberof FeatureLayer
     */
    popupEnabled = true;
    /**
     * 刷新间隔，单位为分钟。默认值为0。0表示不刷新。
     * 最小间隔为0.1分钟。
     *
     * @date 10/12/2021
     * @memberof FeatureLayer
     */
    refreshInterval = 0;
    /**
     * 对于像素单位的元素(icon、labels等)，是否开启透视投影。
     *
     * @date 10/12/2021
     * @memberof FeatureLayer
     */
    screenSizePerspectiveEnabled = true;
}
