import { LayerType, IGSSceneSublayerType } from '../document/layer';
import { CoordinateTransformation } from './coordinate-transformation';
import * as Objects from '../objects';
import { CoordinateSystemType } from '../document/spatial-reference';
import ComputeZoomOffset from '../../utils/map-resolution-util.js';
class FitBound {
    // 记录cesium相机高度与层级对应关系的数组
    _cesiumLevelToHeights = [];
    get cesiumLevelToHeights() {
        return this._cesiumLevelToHeights;
    }
    /**
     * 计算该相机距离距离对应的层级与分辨率
     * @param distance 相机距离该Tile的真实距离
     * @param viewer cesium对象
     * @returns 该相机距离距离对应的层级与分辨率
     */
    detectZoomLevel(distance, viewer) {
        const scene = viewer.scene;
        const tileProvider = scene.globe._surface.tileProvider;
        const quadtree = tileProvider._quadtree;
        // 整个屏幕像素高
        const drawingBufferHeight = viewer.canvas.height;
        // sseDenominator是相机fovy角度的tan值的2倍
        const sseDenominator = viewer.camera.frustum.sseDenominator;
        for (let level = 0; level <= 24; level++) {
            // maxGeometricError是地球赤道的周长/像素数，一个像素代表多少米(该层级下最大的几何误差)
            const maxGeometricError = tileProvider.getLevelMaximumGeometricError(level);
            /** =========以下是公式(maxGeometricError * drawingBufferHeight) / (distance * sseDenominator) 的分解=============== */
            // 根据三角函数的公式可以算得相机距离屏幕中心的像素距离
            const far = drawingBufferHeight / sseDenominator;
            // 相机距离屏幕中心的米单位距离（垂直距离）
            const L = maxGeometricError * far;
            //  sse 屏幕空间误差
            const error = L / distance;
            // 如果屏幕空间误差小于maximumScreenSpaceError最大屏幕空间误差，则返回当前层级
            if (error < quadtree.maximumScreenSpaceError) {
                return level;
            }
        }
        return null;
    }
    /**
     * 获取level与相机高度关系数组
     * @param viewer
     * @returns level与相机高度关系数组
     */
    getLevelDetails(viewer) {
        // 如果_cesiumLevelToHeights高度与层级关系已经计算出来了，则直接返回，不用多次计算
        if (this._cesiumLevelToHeights.length <= 0) {
            /**
             * 这里由于高度达到亿级别，普通循环计算量非常大，这里吧高度分为三段分别采用二分法来计算，提高计算速度
             * @修改人 龚瑞强
             * @date 2021/8/27
             */
            const largeDistances = this.getLevelByStep({
                step: 100000,
                initZoomLevel: 0,
                initHeight: 100000000,
                viewer
            });
            const normalDistances = this.getLevelByStep({
                step: 100,
                initZoomLevel: largeDistances[largeDistances.length - 1].level,
                initHeight: largeDistances[largeDistances.length - 1].minHeight,
                viewer
            });
            const littleDistances = this.getLevelByStep({
                step: 1,
                initZoomLevel: normalDistances[normalDistances.length - 1].level,
                initHeight: normalDistances[normalDistances.length - 1].minHeight,
                viewer
            });
            const levelHeights = [
                ...largeDistances,
                ...normalDistances,
                ...littleDistances
            ];
            // 由于这里设置的高度是最小高度，需要给增加一些高度（这里增加的值是最大高度与最小高度的四分之一），才能保证此高度在该层级范围内
            const arr = [];
            for (let index = 0; index < levelHeights.length; index++) {
                const element = levelHeights[index];
                if (index > 0) {
                    const nearHeight = element.height +
                        (levelHeights[index - 1].height - element.height) / 4;
                    arr.push({
                        ...element,
                        height: nearHeight
                    });
                }
                else {
                    arr.push(element);
                }
            }
            this._cesiumLevelToHeights = arr;
        }
        return this._cesiumLevelToHeights;
    }
    /**
     * 计算层级与最低高度的数组
     * @param step 步进器
     * @param initZoomLevel 初始层级
     * @param initHeight 初始高度
     * @param viewer 三维对象
     * @returns 层级与最低高度的数组
     */
    getLevelByStep({ step, initZoomLevel, initHeight, viewer }) {
        const result = [];
        let currentZoomLevel = initZoomLevel;
        for (let height = initHeight; height >= 0; height -= step) {
            let currentHeight = height;
            let preHeight = height + step;
            /**
             * @description 当此次算出来的层级与上次算出来的层级不同，则采用去上次高度与此次高度的中间值进行计算，
             * 直到上次高度与此次高度差值只有1的时候，则可以计算出来临界值。这样采用二分法计算速度比普通的循环速度快很多
             */
            let level = this.detectZoomLevel(currentHeight, viewer);
            let preLevel = this.detectZoomLevel(preHeight, viewer);
            if (level !== null &&
                level !== undefined &&
                level !== currentZoomLevel &&
                preLevel === currentZoomLevel) {
                while (preHeight - currentHeight > 1) {
                    level = this.detectZoomLevel(currentHeight, viewer);
                    // 当此次算出来的层级与上次算出来的层级不同，则认为当前高度为该层级最小高度
                    preLevel = this.detectZoomLevel(preHeight, viewer);
                    if (level !== null &&
                        level !== undefined &&
                        level !== currentZoomLevel &&
                        preLevel === currentZoomLevel) {
                        preHeight = currentHeight + (preHeight - currentHeight) / 2;
                    }
                    else if (level !== null &&
                        level !== undefined &&
                        level !== currentZoomLevel &&
                        preLevel !== currentZoomLevel) {
                        // 如果两次算得的层级一样，中间值偏大。需要把currentHeight设为此次的高度，然后继续使用二分法进行查找
                        const num = preHeight - currentHeight;
                        currentHeight = preHeight;
                        preHeight = currentHeight + num;
                    }
                }
                result.push({
                    level: currentZoomLevel,
                    height: parseInt(preHeight),
                    minHeight: parseInt(preHeight)
                });
                currentZoomLevel++;
            }
        }
        return result;
    }
    /**
     * 通过层级获取相机高度
     * @param fitLevel 适合的层级
     * @param viewer
     * @returns 相机高度
     */
    getCameraHeightByLevel(fitLevel, viewer) {
        fitLevel = fitLevel <= 2 ? 2 : fitLevel;
        this.getLevelDetails(viewer);
        return this._cesiumLevelToHeights.find((x) => x.level === fitLevel)?.height;
    }
    /**
     * @param bound 范围
     * @param mapParams 地图对象参数
     * @param level 是否带初始缩放级别
     */
    fitBound2D(bound, mapParams, level) {
        const { xmin, ymin, xmax, ymax } = bound;
        const { map } = mapParams;
        // TODO mapboxgl在处理+- 85以外的纬度异常,临时处理，等@mapgis/mapbox-gl更新后去除
        if (ymin < -85 && ymax > 85) {
            level = level || 1.25;
        }
        if (level !== undefined) {
            map.flyTo({
                center: [(xmin + xmax) / 2, (ymin + ymax) / 2],
                zoom: level
            });
        }
        else {
            map.fitBounds([
                [xmin, ymin],
                [xmax, ymax]
            ]);
        }
    }
    /**
     * @param bound 范围
     * @param mapParams 地图对象参数
     * @param level 是否带初始缩放级别
     */
    fitBound3D(bound, mapParams, level) {
        const { xmin, ymin, xmax, ymax } = bound;
        const { viewer, Cesium } = mapParams;
        // 如果获取不到对应层级的高度则还是用范围缩放
        if (level !== undefined) {
            const cameraHeight = this.getCameraHeightByLevel(level, viewer);
            if (cameraHeight !== undefined) {
                const center = new Cesium.Cartesian3.fromDegrees((xmin + xmax) / 2, (ymin + ymax) / 2, cameraHeight);
                viewer.camera.flyTo({
                    destination: center
                });
            }
        }
        else {
            const rectangle = new Cesium.Rectangle.fromDegrees(xmin, ymin, xmax, ymax);
            viewer.camera.flyTo({
                destination: rectangle
            });
        }
    }
    /**
     * 跳转
     * @param layer 图层对象
     * @param mapParams 地图对象
     * @param is2DMap 当前地图模式
     * @param layerExtent 自定义的图层范围（目前只给数据流图层使用）
     * @returns 图层范围是否超出经纬度范围
     */
    fitBoundByLayer(layer, mapParams, is2DMap, layerExtent) {
        const { viewer, map, Cesium, vueCesium } = mapParams;
        const sceneController = Objects.SceneController.getInstance(Cesium, vueCesium, viewer);
        const { type } = layer;
        let fullExtent = null;
        let fullExtentJWD = {
            xmin: 0.0,
            ymin: 0.0,
            xmax: 0.0,
            ymax: 0.0
        };
        // 1.获取图层全图范围
        if (type !== LayerType.IGSScene) {
            /**
             * @修改说明 针对数据流图层直接用传入的范围
             * @修改人 龚瑞强
             * @date 2021/11/23
             */
            if (type === LayerType.DataFlow) {
                fullExtent = layerExtent;
            }
            else {
                // 1.1 非三维图层直接从图层上取范围
                fullExtent = layer.fullExtent;
            }
        }
        else {
            // 1.2 三维图层的范围需要单独计算范围
            // 1.2.1 三维模型缓存子图层中记录的是模型在局部坐标系下的范围,需要转换为经纬度下的范围。
            // 1.2.2 三维地形子图层直接采用子图层中记录的范围。
            const activeScene = layer.activeScene;
            if (activeScene) {
                let range = null;
                activeScene.sublayers.forEach((item) => {
                    switch (item.type) {
                        case IGSSceneSublayerType.elevation:
                            range = item.range;
                            break;
                        case IGSSceneSublayerType.modelCache:
                            range = sceneController.layerExtentToGlobelExtent(item, activeScene.sceneMode);
                            break;
                        default:
                            break;
                    }
                });
                fullExtent = range;
            }
        }
        // 2.将全图范围统一转换为经纬度
        if (layer.spatialReference &&
            layer.spatialReference.wkid === CoordinateSystemType.webMercator) {
            const xminYminConverted = CoordinateTransformation.mercatorToWGS84([
                fullExtent.xmin,
                fullExtent.ymin
            ]);
            const xmaxYmaxConverted = CoordinateTransformation.mercatorToWGS84([
                fullExtent.xmax,
                fullExtent.ymax
            ]);
            fullExtentJWD.xmin = xminYminConverted[0];
            fullExtentJWD.ymin = xminYminConverted[1];
            fullExtentJWD.xmax = xmaxYmaxConverted[0];
            fullExtentJWD.ymax = xmaxYmaxConverted[1];
        }
        else {
            fullExtentJWD = fullExtent;
        }
        const isOutOfRange = this.outOfRange(fullExtentJWD);
        if (isOutOfRange) {
            fullExtentJWD = {
                xmax: 180,
                xmin: -180,
                ymax: 90,
                ymin: -90
            };
        }
        // 3.根据图层类型的不同,进行不同的复位操作。
        // a.瓦片图层(IGSTile、ArcGISTile、OGCWMTS)将视图跳转到以图层全图范围为中心，以尽量能完全显示瓦片的级别为级别的视图下。
        // b.其它图层，将当前视图跳转到以图层全图范围为范围的视图下。
        let fitLevel;
        switch (type) {
            case LayerType.ArcGISTile:
            case LayerType.IGSTile:
                fitLevel = this.getFitZoom(layer.tileInfo, fullExtent, mapParams, is2DMap);
                if (is2DMap) {
                    this.fitBound2D(fullExtentJWD, mapParams, fitLevel);
                }
                else {
                    this.fitBound3D(fullExtentJWD, mapParams, fitLevel);
                }
                break;
            case LayerType.OGCWMTS:
                const tileMatrixSet = layer.activeLayer.tileMatrixSet;
                fitLevel = this.getFitZoom(tileMatrixSet.tileInfo, fullExtent, mapParams, is2DMap);
                if (is2DMap) {
                    this.fitBound2D(fullExtentJWD, mapParams, fitLevel);
                }
                else {
                    this.fitBound3D(fullExtentJWD, mapParams, fitLevel);
                }
                break;
            default:
                if (is2DMap) {
                    this.fitBound2D(fullExtentJWD, mapParams);
                }
                else {
                    this.fitBound3D(fullExtentJWD, mapParams);
                }
                break;
        }
        return isOutOfRange;
    }
    /**
     * 通过范围计算分辨率，通过分辨率查找瓦片图层最适合的显示级别
     * @param tileInfo 瓦片信息
     * @param fullExtent  图层范围
     * @param mapParams
     * @param is2DMap 是否为二维
     * @returns 瓦片图层最适合的显示级别
     */
    getFitZoom(tileInfo, fullExtent, mapParams, is2DMap) {
        let fitLevel = tileInfo.lods[0].levelValue;
        // 1.通过数据范围计算得到对应的分辨率
        let resolution = this.getResolutionForExtent(fullExtent, mapParams, is2DMap);
        if (is2DMap) {
            // 2.mapboxgl地图引擎显示瓦片的大小固定为512，如果数据是256的瓦片，则会被拉伸显示，分辨率会降低。
            // 比如数据是256的瓦片，数据第9级的图片，在mapboxgl里会被拉伸，看起来像是第10级图片对应的范围(分辨率降低了)。
            // 如果想在mapboxgl里看第9级图片对应的范围，则只能拿第8级的数据去显示。
            const tileSize = tileInfo.size[0];
            const scale = tileSize / 512;
            // 3.计算在mapboxgl里显示的实际分辨率
            const mapboxglResolution = resolution / scale;
            resolution = mapboxglResolution;
        }
        // 4.使用mapboxgl里显示的实际分辨率去查找数据对应级别
        if (tileInfo.lods.length > 1) {
            for (let i = 0; i < tileInfo.lods.length - 1; i++) {
                const iResolution = tileInfo.lods[i].resolution;
                if (i == 0) {
                    // 如果分辨率比第一个分辨率大，则直接取第一个的级别
                    if (resolution > tileInfo.lods[i].resolution) {
                        fitLevel = tileInfo.lods[i].level;
                    }
                }
                else if (i == tileInfo.lods.length - 2) {
                    // 如果分辨率比最后一个分辨率小，则直接取最后一个的级别，这里i最大能取到tileInfo.lods.length - 2
                    if (resolution < tileInfo.lods[i].resolution) {
                        fitLevel = tileInfo.lods[i + 1].level;
                    }
                }
                else {
                    // 级别向下取，比如，如果分辨率对应级别是8.5，则取级别8
                    if (resolution < tileInfo.lods[i].resolution &&
                        resolution > tileInfo.lods[i + 1].resolution) {
                        fitLevel = tileInfo.lods[i].level;
                    }
                }
            }
        }
        // 由于三维ArcGISTile、IGSTile没有设置瓦片的偏移量，所以这里不需要计算
        if (is2DMap) {
            const zoomOffset = ComputeZoomOffset.getZoomOffsetByTileInfo(tileInfo, !is2DMap);
            // 修改说明：由于Tile设置了偏移量zoomOffset,为了保证刚好缩放到Tile的起始级别，需要将fitLevel-zoomOffset
            fitLevel -= zoomOffset;
        }
        return fitLevel;
    }
    /**
     * 范围是否超出最大经纬度范围
     * @param fullExtentJWD
     * @returns 范围是否超出最大经纬度范围
     */
    outOfRange(fullExtentJWD) {
        const { ymax, ymin, xmax, xmin } = fullExtentJWD;
        if (ymax && ymin && xmax && xmin) {
            return (Number(xmax) < -180 ||
                Number(xmax) > 180 ||
                Number(xmin) < -180 ||
                Number(xmin) > 180 ||
                Number(ymax) < -90 ||
                Number(ymax) > 90 ||
                Number(ymin) < -90 ||
                Number(ymin) > 90);
        }
        return false;
    }
    /**
     * 根据范围计算对应的分辨率
     * @param bound 图层范围
     * @param mapParams
     * @param is2DMap 是否是二维
     * @returns 分辨率
     */
    getResolutionForExtent(bound, mapParams, is2DMap) {
        const { viewer, map, Cesium } = mapParams;
        const devicePixelRatio = window.devicePixelRatio || 1;
        let width;
        let height;
        if (is2DMap) {
            width = parseInt(map.getCanvas().style.width) * devicePixelRatio;
            height = parseInt(map.getCanvas().style.height) * devicePixelRatio;
        }
        else {
            width =
                parseInt(viewer.canvas.width) ||
                    parseInt(viewer.container.offsetWidth) * devicePixelRatio;
            height =
                parseInt(viewer.canvas.height) ||
                    parseInt(viewer.container.offsetHeight) * devicePixelRatio;
        }
        const { xmin, ymin, xmax, ymax } = bound;
        const xlength = xmax - xmin;
        const ylength = ymax - ymin;
        const length = xlength - ylength > 0 ? xlength : ylength;
        const xResolution = length / width;
        const yResolution = length / height;
        // console.log(xResolution)
        // console.log(yResolution)
        return xResolution - yResolution < 0 ? yResolution : xResolution;
    }
    /**
     * 根据分辨率计算范围
     * @param bound 图层范围
     * @param mapParams
     * @param resolution 分辨率
     * @param is2DMap
     * @returns 范围
     */
    getBoundForResolution(bound, mapParams, resolution, is2DMap) {
        const { viewer, map, Cesium } = mapParams;
        const devicePixelRatio = window.devicePixelRatio || 1;
        let width;
        let height;
        if (is2DMap) {
            width = parseInt(map.getCanvas().style.width) * devicePixelRatio;
            height = parseInt(map.getCanvas().style.height) * devicePixelRatio;
        }
        else {
            width =
                parseInt(viewer.canvas.width) ||
                    parseInt(viewer.container.offsetWidth) * devicePixelRatio;
            height =
                parseInt(viewer.canvas.height) ||
                    parseInt(viewer.container.offsetHeight) * devicePixelRatio;
        }
        const { xmin, ymin, xmax, ymax } = bound;
        const xCenter = (xmax + xmin) / 2;
        const yCenter = (ymax + ymin) / 2;
        const xLength = resolution * width;
        const yLength = resolution * height;
        const nXmin = xCenter - xLength / 2;
        const nXmax = xCenter + xLength / 2;
        const nYmin = yCenter - yLength / 2;
        const nYmax = yCenter + yLength / 2;
        // console.log(xmin, ymin, xmax, ymax)
        // console.log(nXmin, nYmin, nXmax, nYmax)
        return [nXmin, nYmin, nXmax, nYmax];
    }
}
export default new FitBound();
