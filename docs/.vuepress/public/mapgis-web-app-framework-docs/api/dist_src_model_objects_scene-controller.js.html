export class SceneController {
    static sceneControllerArray = [];
    Cesium = undefined;
    vueCesium = undefined;
    viewer = undefined;
    static getInstance(Cesium, vueCesium, viewer) {
        let sceneControllerValue = this.sceneControllerArray.find((item) => {
            return item.viewer === viewer;
        });
        if (sceneControllerValue) {
            return sceneControllerValue.sceneController;
        }
        sceneControllerValue = {
            viewer,
            sceneController: new SceneController(Cesium, vueCesium, viewer)
        };
        this.sceneControllerArray.push(sceneControllerValue);
        return sceneControllerValue.sceneController;
    }
    constructor(Cesium, vueCesium, viewer) {
        this.Cesium = Cesium;
        this.vueCesium = vueCesium;
        this.viewer = viewer;
    }
    /**
     * 截图
     * @param imgName 输出图片名
     */
    outputImage(imgName) {
        const { viewer } = this;
        viewer.render();
        const res = this.Cesium.ReImg.fromCanvas(viewer.canvas);
        res.downloadPng(imgName);
    }
    /**
     * rgba或者十六进制转换成cesium.color
     * @param color The CSS color value in #rgb, #rgba, #rrggbb, #rrggbbaa, rgb(), rgba(), hsl(), or hsla() format.
     * @returns rgba或者十六进制转换成cesium.color
     */
    colorToCesiumColor(color) {
        return this.Cesium.Color.fromCssColorString(color);
    }
    /**
     * 获取当前范围
     */
    getCurrentExtent() {
        // 范围对象
        const extent = {
            xmin: -180,
            ymax: 90,
            xmax: 180,
            ymin: -90,
            height: 0
        };
        const rectangle = this.viewer.camera.computeViewRectangle();
        extent.xmin = (rectangle.west / Math.PI) * 180;
        extent.ymax = (rectangle.north / Math.PI) * 180;
        extent.xmax = (rectangle.east / Math.PI) * 180;
        extent.ymin = (rectangle.south / Math.PI) * 180;
        // 获取高度
        extent.height = Math.ceil(this.viewer.camera.positionCartographic.height);
        return extent;
    }
    // 屏幕坐标转地理坐标
    screenPositionToGlobelPosition(mousePoint) {
        // 通过指定的椭球或者地图对应的坐标系，将鼠标的二维坐标转换为对应椭球体三维坐标
        const { ellipsoid } = this.viewer.scene.globe;
        let cartesian;
        if (mousePoint.position) {
            // 拿到屏幕坐标  【屏幕坐标转笛卡尔坐标】
            cartesian = this.viewer.camera.pickEllipsoid(mousePoint.position, ellipsoid);
        }
        else {
            cartesian = this.viewer.camera.pickEllipsoid(mousePoint.endPosition, ellipsoid);
        }
        if (cartesian) {
            // 将笛卡尔坐标转换为地理坐标
            const cartographic = ellipsoid.cartesianToCartographic(cartesian);
            // 将弧度转为度的十进制度表示
            const longitude = this.Cesium.Math.toDegrees(cartographic.longitude); // 转换后的经度
            const latitude = this.Cesium.Math.toDegrees(cartographic.latitude); // 转换后的纬度
            const currentCoor = [longitude, latitude];
            return currentCoor;
        }
        return null;
    }
    /**
     * 地理坐标转局部坐标
     * @param  {object} position 经纬度点
     * @param  {object} transform 对应数据的transform信息
     * @return {Cartographic}  Cartographic 数据坐标
     */
    globelPositionToLocalPosition(position, transform, offset) {
        if (offset == undefined)
            offset = [0, 0, 0];
        // 度转迪卡尔
        const c3 = this.Cesium.Cartesian3.fromDegrees(position.x, position.y, position.z);
        let matrixInver = new this.Cesium.Matrix4();
        matrixInver = this.Cesium.Matrix4.inverse(transform, matrixInver);
        const maxCar = new this.Cesium.Cartesian3();
        this.Cesium.Matrix4.multiplyByPoint(matrixInver, c3, maxCar);
        const localPos = new this.Cesium.Cartesian3(maxCar.x - offset[0], maxCar.y - offset[1], maxCar.z);
        return localPos;
    }
    /**
     * 数据局部坐标转经纬度地理坐标
     * @param  {object} position 数据坐标点
     * @param  {object} transform 对应数据的transform信息
     * @return {object}  Position 经纬度坐标
     */
    localPositionToGlobelPosition(position, transform, offset) {
        if (offset == undefined)
            offset = [0, 0, 0];
        let car3 = new this.Cesium.Cartesian3();
        const pos = {
            x: position.x + offset[0],
            y: position.y + offset[1],
            z: position.z
        };
        car3 = this.Cesium.Matrix4.multiplyByPoint(transform, pos, car3);
        const cartographic = this.Cesium.Cartographic.fromCartesian(car3);
        const lonDegree = this.Cesium.Math.toDegrees(cartographic.longitude);
        const latDegree = this.Cesium.Math.toDegrees(cartographic.latitude);
        const heightDegree = cartographic.height; // this.Cesium.Math.toDegrees(radiusPoint.height);鼠标Z值代表真实Z值
        const positions = { x: lonDegree, y: latDegree, z: heightDegree };
        return positions;
    }
    /**
     * 数据局部坐标范围转经纬度地理坐标范围
     * @param  {object} extent3D 数据坐标点范围
     * @param  {object} transform 对应数据的transform信息
     * @return {object}  Position 经纬度坐标
     */
    localExtentToGlobelExtent(extent3D, transform, offset) {
        const centerMinPoint = new this.Cesium.Cartesian3();
        centerMinPoint.x = extent3D.xmin;
        centerMinPoint.y = extent3D.ymin;
        centerMinPoint.z = extent3D.zmin;
        const positions1 = this.localPositionToGlobelPosition(centerMinPoint, transform, offset); // 数据坐标转经纬度坐标
        const centerMaxPoint = new this.Cesium.Cartesian3();
        centerMaxPoint.x = extent3D.xmax;
        centerMaxPoint.y = extent3D.ymax;
        centerMaxPoint.z = extent3D.zmax;
        const positions2 = this.localPositionToGlobelPosition(centerMaxPoint, transform, offset); // 数据坐标转经纬度坐标
        const bound = {
            xmin: positions1.x,
            ymin: positions1.y,
            xmax: positions2.x,
            ymax: positions2.y
        };
        return bound;
    }
    /**
     * 世界坐标范围转经纬度范围
     * @param extent3D 世界坐标系范围
     * @returns 经纬度范围
     */
    worldExtentToGlobelExtent(extent3D) {
        const ellipsoid = this.viewer.scene.globe.ellipsoid;
        const minPoint = new this.Cesium.Cartesian3();
        minPoint.x = extent3D.xmin;
        minPoint.y = extent3D.ymin;
        minPoint.z = extent3D.zmin;
        const cartographicMin = ellipsoid.cartesianToCartographic(minPoint);
        const latMin = this.Cesium.Math.toDegrees(cartographicMin.latitude);
        const lngMin = this.Cesium.Math.toDegrees(cartographicMin.longitude);
        const maxPoint = new this.Cesium.Cartesian3();
        maxPoint.x = extent3D.xmax;
        maxPoint.y = extent3D.ymax;
        maxPoint.z = extent3D.zmax;
        const cartographicMax = ellipsoid.cartesianToCartographic(maxPoint);
        const latMax = this.Cesium.Math.toDegrees(cartographicMax.latitude);
        const lngMax = this.Cesium.Math.toDegrees(cartographicMax.longitude);
        const bound = {
            ymin: latMin,
            xmin: lngMax,
            ymax: latMax,
            xmax: lngMax
        };
        return bound;
    }
    /**
     * 图层数据局部坐标范围转经纬度地理范围
     * @param igsSceneSubLayer IGS三维图层
     * @param vueKey 可选，对应viewer的key
     * @param sceneMode 场景模式:2->平面模式（对应局部坐标系）、1->球面面模式（对应世界坐标系）
     * @retu rns 经纬度坐标
     */
    layerExtentToGlobelExtent(igsSceneSubLayer, sceneMode, vueKey) {
        const { id, range } = igsSceneSubLayer;
        let bound = null;
        switch (sceneMode) {
            case 1:
                bound = this.worldExtentToGlobelExtent(range);
                break;
            case 2:
                /**
                 * @修改说明 现在新版viewer已经携带了vueKey，再不需要传入vueKey了
                 * @修改人 龚瑞强
                 * @修改时间 2021/7/5
                 */
                /**
                 * @修改说明 兼容传入的vueKey，默认取viewer上的
                 * @修改人 龚跃健
                 * @修改时间 2021/11/1
                 */
                const source = this.findSource(id, vueKey);
                if (source) {
                    const tranform = source.root.transform;
                    bound = this.localExtentToGlobelExtent(range, tranform);
                }
                break;
            default:
                break;
        }
        return bound;
    }
    /**
     * 获取M3DLayer对象
     * @returns M3DLayer对象
     */
    getM3D() {
        /**
         * @修改说明 TODO cesium1.8暂时没有获取M3D的接口,暂时使用window.CesiumZondy
         * @修改人 龚跃健
         * @修改时间 2021/11/8
         */
        const { vueCesium, viewer } = this;
        const m3d = new window.CesiumZondy.Layer.M3DLayer({
            viewer
        });
        return m3d;
    }

    /**
     * 查找模型对象
     * @param id 查找模型的id
     * @param vueKey 可选，对应viewer的key
     * @returns 模型对象
     */
    findSource(id, vueKey) {
        vueKey = vueKey || this.viewer.vueKey;
        let sourceId = id;
        let layerIndex = 0;
        if (id.includes(':')) {
            sourceId = id.split(':')[0];
            layerIndex = Number(id.split(':')[1]);
        }
        const { options } = this.vueCesium.G3DManager.findSource(vueKey, sourceId) || {};
        if (options && options.m3ds && options.m3ds.length > 0) {
            return options.m3ds[layerIndex];
        }
        return undefined;
    }
    /**
     * 查找M3D对象
     * @param id 查找模型的id
     * @param vueKey 可选，对应viewer的key
     * @returns 模型对象

    findM3DIgsSource(id, vueKey) {
        vueKey = vueKey || this.viewer.vueKey;
        let sourceId = id;
        let layerIndex = 0;
        if (id.includes(':')) {

            layerIndex = Number(id.split(':')[1]);
        }
        const { source } = this.vueCesium.M3DIgsManager.findSource(vueKey, sourceId) || {};
        if (source) {
            return source[layerIndex];
        }
        return undefined;
    }

    /**
     * 获取三维经纬度范围
     * @returns 三维经纬度范围
     */
    getComputeViewRectangle() {
        const { west, north, east, south } = this.viewer.camera.computeViewRectangle();
        const xmin = (west / Math.PI) * 180;
        const ymax = (north / Math.PI) * 180;
        const xmax = (east / Math.PI) * 180;
        const ymin = (south / Math.PI) * 180;
        return {
            xmin,
            ymin,
            xmax,
            ymax
        };
    }

     * @returns 地理坐标

        const pickEllipsoid = this.getPickEllipsoid();
        return pickEllipsoid
            ? this.Cesium.Ellipsoid.WGS84.cartesianToCartographic(pickEllipsoid)
            : null;
    }
    /**
     * 获取camera高度
     * @returns height
     */
    getCameraHeight() {
        const { viewer } = this;
        return viewer
            ? viewer.scene.globe.ellipsoid.cartesianToCartographic(viewer.camera.position).height
            : 0;
    }
    /**
     * 获取当前摄像机的位置
     * @returns 当前摄像机的位置
     */
    getCenterPosition() {
        const lnglat = this.getCartographic();
        if (lnglat) {
            const { longitude, latitude } = lnglat;
            const pi = Math.PI;
            const lng = (longitude * 180) / pi;
            const lat = (latitude * 180) / pi;
            const height = this.getCameraHeight();
            return {
                lng,
                lat,
                height
            };
        }
        return null;
    }

    }

        const { enabled, uniforms } = this.viewer.scene.brightness;

        return { enabled, brightness };
    }
    /**

        const sceneBrightness = this.viewer.scene.brightness;
        sceneBrightness.enabled = enabled;
        sceneBrightness.uniforms.brightness = brightness;
    }
