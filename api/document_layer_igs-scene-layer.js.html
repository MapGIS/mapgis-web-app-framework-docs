    checkServiceVersion(url) {
        const serverType = 'igs/rest/g3d/';
        const indexServer = url.search(serverType);
        if (indexServer === -1) {
            return 'sceneServer';

        return 'g3d';

    load() {
        // 只有加载状态是没有加载过时，才会真正进行请求。
        if (this.loadStatus !== LoadStatus.notLoaded) {
            return new Promise((resolve) => {
                resolve(this);
            }).then((data) => {
                return this;
            });
        }
        const serviceVersion = this.checkServiceVersion(this.url);
        if (serviceVersion === 'sceneServer') {
            // sceneServer服务
            // 1.拼接服务地址
            // 获取场景信息的url
            const sceneInfoUrl = `${this.url}?f=json`;
            // 获取场景子图层的url
            const layerInfoUrl = `${this.url}/layers?f=json`;
            const promise = new Promise((resolve, reject) => {
                this.loadStatus = LoadStatus.loading;
                // 2.请求场景信息
                // 2.1 发请求
                axios.get(sceneInfoUrl).then((res) => {
                    const { data } = res;
                    if (!data) {
                        resolve(undefined);
                    }
                    else {
                        // 2.2 解析请求结果
                        const scene = new Scene();
                        scene.layer = this;
                        scene.fromJSON(data);
                        this.scenes.push(scene);
                        if (this.scenes.length > 0)
                            this.activeScene = this.scenes[0];
                        resolve(this);
                    }
                }, (error) => {
                    this.loadStatus = LoadStatus.failed;
                    reject(error);
                });
            });
            return promise.then((sceneLayer) => {
                const promise = new Promise((resolve, reject) => {
                    // 3.请求图层信息
                    // 3.1 发请求
                    axios.get(layerInfoUrl).then((res) => {
                        const { data } = res;
                        if (data) {
                            // 3.2 解析请求结果
                            this.activeScene?.fromJSON(data);
                            this.loadStatus = LoadStatus.loaded;
                            resolve(this);
                        }
                    }, (error) => {
                        this.loadStatus = LoadStatus.failed;
                        reject(error);
                    });
                });
                return promise.then((sceneLayer) => {
                    return this;
                });
            });
        }
        else {
            // G3D服务
            // 1.从URL中解析出ip、port、serverName参数
            const parmsBase = this._parseUrl(this.url);
            // 1.1 扩展属性的key列表参数
            const parms = { ...parmsBase, extendedPropKeys: this.extendedPropKeys };
            const promise = new Promise((resolve, reject) => {
                // 2.调用es6service中的接口进行网络请求。
                const g3dMapDoc = new G3D.G3DMapDoc(parms);
                this.loadStatus = LoadStatus.loading;
                g3dMapDoc.GetDocInfo((res) => {
                    if (res && res.sceneInfos) {
                        if (res.sceneInfos.length > 0) {
                            res.sceneInfos.forEach((sceneInfo) => {
                                const scene = new Scene();
                                scene.layer = this;
                                scene.fromJSON(sceneInfo);
                                this.scenes.push(scene);
                            });
                        }
                    }
                    if (this.scenes.length > 0)
                        this.activeScene = this.scenes[0];
                    this.loadStatus = LoadStatus.loaded;
                    resolve(this);
                }, (failInfo) => {
                    this.loadStatus = LoadStatus.failed;
                    reject(failInfo);
                });
            });
            return promise.then((data) => {
                return this;
            });
        }
    }
    clone() {
        const result = new IGSSceneLayer();
        Object.entries(this).forEach((element) => {
            const key = element[0];
            const valueIndex = 1;
            if (key === 'activeScene') {
            }
            else if (key === 'scenes') {
                const scenes = element[valueIndex];
                const scenesCopy = [];
                let sceneCopy;
                scenes.forEach((scene) => {
                    sceneCopy = scene.clone(result);
                    if (sceneCopy) {
                        scenesCopy.push(sceneCopy);
                    }
                });
                result[key] = scenesCopy;
            }
            else {
                result[key] = ObjectTool.deepClone(element[valueIndex]);
            }
        });
        let avtiveScene;
        result.scenes.forEach((scene) => {
            if (scene.sceneIndex === this.activeScene?.sceneIndex)
                avtiveScene = scene;
        });
        result.activeScene = avtiveScene;
        return result;
    }
    /**
     * url解析,提取对应的ip、port、docName
     * 约定URL格式如下：[protocol]://[ip]:[port]/igs/rest/g3d/{docName}
     *
     * @date 23/03/2021
     * @private
     * @param {*} url
     * @return {*}  {{ ip; port; docName }}
     * @memberof IGSTileLayer
     */
    _parseUrl(url) {
        const ipReg = '/://[a-zA-Z0-9]+:*/g';
        const portReg = '/:+[0-9]+//g';
        const serverType = 'igs/rest/g3d/';
        const indexServer = url.search(serverType);
        const indexName = indexServer + serverType.length;
        const docName = url.substr(indexName);
        const ips = url.match(/:\/\/[a-zA-Z0-9.]+:*/g);
        const ports = url.match(/:+[0-9]+\//g);
        const matchIp = ips ? ips[0] : '://localhost';
        const matchPort = ports ? ports[0] : ':6163';
        let ip = '';
        let port = '';
        if (matchIp && matchIp.length > 3) {
            ip = matchIp.slice(3, matchIp.length - 1);
        }
        if (matchPort && matchPort.length > 2) {
            port = matchPort.slice(1, matchPort.length - 1);
        }
        return { ip, port, docName };
    }

    fromJSON(jsonObject) {
        if (jsonObject.layerRenderIndex !== undefined)
            this.layerIndex = jsonObject.layerIndex;
        if (jsonObject.layerName !== undefined)
            this.title = jsonObject.layerName;
        if (jsonObject.gdbpUrl !== undefined)
            this.url = jsonObject.gdbpUrl;
        if (jsonObject.isVisible !== undefined)
            this.visible = jsonObject.isVisible;
        if (jsonObject.layerType !== undefined)
            this.type = parseInt(jsonObject.layerType); // 字符转数字。
        if (jsonObject.layerRenderType !== undefined)
            this.renderType = jsonObject.layerRenderType;
        if (jsonObject.range3D !== undefined) {
            /**
             * @修改说明 这里为了处理高斯动态投影到经纬度的情况，range3D任然是原始坐标，这里使用range的坐标作为替代

            this.range = new Rectangle3D(jsonObject.range.xMin, jsonObject.range.yMin, jsonObject.range3D.zMin, jsonObject.range.xMax, jsonObject.range.yMax, jsonObject.range3D.zMax);
        }
        if (jsonObject.beginLevel !== undefined)
            this.beginLevel = jsonObject.beginLevel;
        if (jsonObject.endLevel !== undefined)
            this.endLevel = jsonObject.endLevel;
        if (jsonObject.originalPoints) {
            this.originPoint = new Point3D(jsonObject.originalPoints.x, jsonObject.originalPoints.y, 0);
        }
        if (jsonObject.frameSizes && jsonObject.frameSizes.length > 0) {
            this.maxFrameSize = new Size(jsonObject.frameSizes[0].wid, jsonObject.frameSizes[0].hel);
        }
        if (jsonObject.extendedProps) {
            this.extendedProps = jsonObject.extendedProps;
        }
        if (this.layer)
            this.id = `${this.layer.id}:${this.layerIndex}`;
    }
    /**
     * 将该对象转换为json对象
     *
     * @date 30/03/2021
     * @return {*}  {Record<string, any>}
     * @memberof IGSSceneSublayer
     */
    toJSON() {
        return {};
    }
    /**
     * 创建一个深度克隆的sublayer
     *
     * @date 02/04/2021
     * @return {*}  {IGSSceneSublayer}
     * @memberof IGSSceneSublayer
     */
    clone(scene) {
        const result = new IGSSceneSublayer();
        Object.entries(this).forEach((element) => {
            const key = element[0];
            const valueIndex = 1;
            if (key === 'layer') {
                if (scene)
                    result[key] = scene.layer;
            }
            else {
                result[key] = ObjectTool.deepClone(element[valueIndex]);
            }
        });
        return result;
    }
